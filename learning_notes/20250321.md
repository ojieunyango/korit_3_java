int a = 10;
a += 5;  // a = a + 5 (a = 15)
System.out.println(a);  // 출력: 15

a -= 3;  // a = a - 3 (a = 12)
System.out.println(a);  // 출력: 12

a *= 2;  // a = a * 2 (a = 24)
System.out.println(a);  // 출력: 24

a /= 4;  // a = a / 4 (a = 6)
System.out.println(a);  // 출력: 6

a %= 4;  // a = a % 4 (a = 2)
System.out.println(a);  // 출력: 2
